# Docker install
curl -fsSL https://get.docker.com -o install-docker.sh
sudo sh install-docker.sh

# Docker mirror
sudo gnome-text-editor /etc/docker/daemon.json
{
    "registry-mirrors": [
        "https://docker.m.daocloud.io",
        "https://docker.1panel.live",
        "https://hub.rat.dev"
    ],
    "features": {
        "containerd-snapshotter": true
    }
}
sudo systemctl daemon-reload
sudo systemctl restart docker

# Docker ubuntu
sudo docker create -it --name aa ubuntu
sudo docker start aa
sudo docket exec aa ls
sudo docker attach aa
sudo docker stop aa

# Docker command
sudo docker images
sudo docker ps -a
sudo docker rm
sudo docker rmi
docker system prune -a
docker create -v entrypoint:/entrypoint aa

# Docker vscode
sudo usermod -aG docker $USER
newgrp docer
reboot

# Docker apt
echo "deb http://mirrors.aliyun.com/ubuntu/ focal main restricted universe multiverse" > /etc/apt/sources.list
